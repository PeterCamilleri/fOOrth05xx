    class XfOOrth::Context

    The Context class is used to maintain information about the compilation process as it proceeds.

    Attributes

    buffer[R]

    The compile text buffer.

    info[R]

    The context variables.

    sym_map[R]

    Get the symbol mapping in effect for this context.

def unnest(tags)  # ??? WIP rename to unnest_context
  verify_tag(tags)
  @contexts.pop

  if @contexts.empty?
    @buffer << '}'
    eval(@buffer)
  else
    nil
  end

def verify_tag(tags)
  ferr "No valid context." if @contexts.empty?
  tag = @contexts[-1][0]

  unless tags.include?(tag)
    ferr "Error: Found #{tag} but expected #{tags.inspect}"
  end

  tag
end

def initialize(string_list)    
  reset_read_point
  
  if string_list.is_a?(String)
    @string_list = [ string_list ]
  else
    @string_list = string_list
  end
  
  @read_step = @string_list.each
  @eof       = false
end

def get
  return nil if @eof
  
  read do
    begin
      @read_step.next.rstrip
    rescue StopIteration
      @eof = true
      nil
    end
  end
end

def test_that_it_handles_modes
  assert_equal(:Execute, @c.mode)
  @c.open('test', mode: :Compile)
  assert_equal(:Compile, @c.mode)
  @c.nest('[', mode: :Execute)
  assert_equal(:Execute, @c.mode)
  @c.unnest(['['])
  assert_equal(:Compile, @c.mode)
  @c.close(['test'])
  assert_equal(:Execute, @c.mode)
end
